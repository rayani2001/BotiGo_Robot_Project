How BotiGo Works

BotiGo can be controlled in three different ways.

1. Obstacle Avoidance Mode

In this mode, BotiGo moves autonomously while avoiding obstacles.

  • The ultrasonic sensor continuously scans the environment
  • When an obstacle is detected within the threshold distance (usually 15-20cm), BotiGo stops
  • The servo motor rotates the sensor to scan left and right
  • BotiGo chooses the path with more space and turns accordingly
  • If both sides are blocked, BotiGo reverses and finds a new path

2. Bluetooth Control Mode

Directly control BotiGo using a smartphone app.

  • Connect your smartphone to BotiGo via Bluetooth
  • Open a robot controller app (like Arduino Bluetooth Controller)
  • Use the directional buttons to control movement
  • Forward, backward, left, right, and stop commands are available
  • Speed control options may also be available depending on the app

3. Voice Control Mode

Control BotiGo using voice commands through your smartphone.

  • Connect to BotiGo via Bluetooth
  • Open a voice control app that supports Bluetooth devices
  • Use voice commands like "forward", "back", "turn left", "turn right", "stop"
  • The app converts voice to text commands
  • Commands are sent to BotiGo via Bluetooth
  • The Arduino processes these commands and controls the motors accordingly

Arduino Code - Main Sketch for BotiGo
// BotiGo: Multi-Function Arduino Robot
// Created by: RAYANI MINOLI WARNAKULASURIYA DON FERNANDO | GROUP: 4240
// Features: Obstacle Avoidance, Bluetooth Control, Voice Control

#include <Servo.h>
#include <SoftwareSerial.h>

// Define motor control pins
#define IN1 8
#define IN2 9
#define IN3 10
#define IN4 11
#define ENA 5
#define ENB 6

// Ultrasonic sensor pins
#define TRIG_PIN 3
#define ECHO_PIN 2

// Servo pin
#define SERVO_PIN 12

// Bluetooth module pins
#define BT_RX 0
#define BT_TX 1

// Control mode
#define MODE_OBSTACLE_AVOIDANCE 0
#define MODE_BLUETOOTH 1
#define MODE_VOICE 2

int currentMode = MODE_OBSTACLE_AVOIDANCE;

// Create objects
Servo servoMotor;
SoftwareSerial bluetoothSerial(BT_RX, BT_TX);

void setup() {
  // Initialize motor control pins
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  pinMode(ENA, OUTPUT);
  pinMode(ENB, OUTPUT);
  
  // Initialize ultrasonic sensor pins
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  
  // Initialize servo
  servoMotor.attach(SERVO_PIN);
  servoMotor.write(90); // Center position
  
  // Initialize serial communication
  Serial.begin(9600);
  bluetoothSerial.begin(9600);
  
  delay(2000); // Give time for systems to initialize
}

void loop() {
  // Check for mode changes from Bluetooth
  if (bluetoothSerial.available()) {
    char cmd = bluetoothSerial.read();
    processCommand(cmd);
  }
  
  // Execute current mode
  switch (currentMode) {
    case MODE_OBSTACLE_AVOIDANCE:
      obstacleAvoidanceMode();
      break;
    case MODE_BLUETOOTH:
      bluetoothControlMode();
      break;
    case MODE_VOICE:
      voiceControlMode();
      break;
  }
}

// Function to measure distance using ultrasonic sensor
long measureDistance() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  long duration = pulseIn(ECHO_PIN, HIGH);
  long distance = duration * 0.034 / 2; // Convert to cm
  
  return distance;
}
